{
	int input;

	while( input != -1 )
	{
		cout << "Please input a number or enter -1 to exit: ";
		cin >> input;
		clrscr();

		cout << "Multiplication Table:" << endl << endl;

		for ( int i = 0; i <= 12; i++ )

		cout << "Press enter to continue";
		getch();
		clrscr();
	}

	getch();
	return 0;
}

#include<string>
#include<conio.h>
#include<iostream>

using namespace std;

//----------------------------------------------------------------------------------------------------------------

int main()
{
	string userName;
	string first;
	string last;
	int counter = 0; //This makes sure that the first name entered goes into first position

	do{
		cout << "Please input a name: ";
		cin >> userName;

		if ( counter == 0 )
		{
			first = userName;
			last = userName;
			counter++;
		}

		else
			if ( userName < first ){
				counter++;
				first = userName;
			}

		else
			if ( userName > last ){
			counter++;
			last = userName;
		}
	}
	while ( userName != "exit" );

	clrscr();
	cout << "Information" << endl << endl;

	if ( first == "exit" && last == "exit" )
		cout << "No names were entered";


	else
		if ( counter == 2 && first == "exit" )
			cout << "Only one name entered is:\n" << last;
		

		else
			if ( counter == 2 && last == "exit" )
				cout << "Only one name entered is:\n" << first;
			

			else
				cout << "The first name is " << first << " and the last name is " << last << endl;

	getch();
	return 0;
}
